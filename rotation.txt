#include <GL/glut.h>
#include <math.h>

float angle = 45;   // Rotation angle in degrees

// Original rectangle coordinates
float x1 = 100, y1 = 100;
float x2 = 200, y2 = 200;

// Function to rotate a point
void rotatePoint(float x, float y, float *xr, float *yr)
{
    float rad = angle * 3.1416 / 180.0;

    *xr = x * cos(rad) - y * sin(rad);
    *yr = x * sin(rad) + y * cos(rad);
}

// Function to draw rectangle
void drawRectangle(float x1, float y1, float x2, float y2)
{
    glBegin(GL_LINE_LOOP);
        glVertex2f(x1, y1);
        glVertex2f(x2, y1);
        glVertex2f(x2, y2);
        glVertex2f(x1, y2);
    glEnd();
}

// Display function
void display()
{
    float rx1, ry1, rx2, ry2, rx3, ry3, rx4, ry4;

    glClear(GL_COLOR_BUFFER_BIT);

    // Original Rectangle (Black)
    glColor3f(0.0, 0.0, 0.0);
    drawRectangle(x1, y1, x2, y2);

    // Rotate each vertex
    rotatePoint(x1, y1, &rx1, &ry1);
    rotatePoint(x2, y1, &rx2, &ry2);
    rotatePoint(x2, y2, &rx3, &ry3);
    rotatePoint(x1, y2, &rx4, &ry4);

    // Rotated Rectangle (Red)
    glColor3f(1.0, 0.0, 0.0);

    glBegin(GL_LINE_LOOP);
        glVertex2f(rx1, ry1);
        glVertex2f(rx2, ry2);
        glVertex2f(rx3, ry3);
        glVertex2f(rx4, ry4);
    glEnd();

    glFlush();
}

// Initialization
void init()
{
    glClearColor(1.0, 1.0, 1.0, 1.0);
    gluOrtho2D(-500, 500, -500, 500);
}

// Main function
int main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutCreateWindow("2D Rotation");

    init();
    glutDisplayFunc(display);

    glutMainLoop();
    return 0;
}
